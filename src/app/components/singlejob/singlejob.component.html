ห้องบรรยายห้องปฎิบัติการห้องประชุมอยู่อาคารไหนและชั้นไหนบ้าง
SELECT building.name,buildsystem.class,room.id,room.name
FROM room,buildsystem
INNER JOIN building ON buildsystem.buildding_id = building.id
WHERE room.name LIKE "%ห้องบรรยาย%" OR room.name LIKE "%ห้องประชุม%" OR room.name LIKE "%ห้องปฏิบัติการ%" AND room.id = buildsystem.room_id
GROUP BY room.id
ORDER BY room.name

FROM buildsystem AS st , room AS ro
